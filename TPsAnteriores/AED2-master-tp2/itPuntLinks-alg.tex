\begin{algorithm}[H]
\caption{iBuscarMax}
\begin{algorithmic}[1]
\Function{ibuscarMax}{\param{in}{it}{estrITPL}, \param{in}{f}{Fecha}}{$\ensuremath{\rightarrow}$ \param{}{res}{itPuntLinks}}
\state res $\gets$ copiarIt(it) \hfill //O(1)
\while{haySiguiente(it)} \hfill //O(longitud(siguientes(it)))
\If{cantAccesosDesde(it,f) > cantAccesosDesde(res,f)} \hfill //O(1)
\state res $\gets$ it \hfill //O(1)
\endif
\state avanzar(it) \hfill //O(1)
\endwhile
\EndFunction 
\end{algorithmic}
\hrule
\complejidad{longitud(siguientes(it))}
\end{algorithm}

\begin{algorithm}[H]
\caption{iCantAccesosDesde}
\begin{algorithmic}[1]
\Function{icantAccesosDesde}{\param{in}{it}{estrITPL}, \param{in}{f}{Fecha}}{$\ensuremath{\rightarrow}$ \param{}{res}{nat}}
\state itAccesos itAcc $\gets$ (*siguiente(it)).accesos \hfill //O(1)
\state res $\gets$ 0 \hfill //O(1)
\While{haySiguiente(itAcc)} \hfill //O(1)
\If{(siguiente(itAcc)).dia $\leq$ f $\wedge$ (siguiente(itAcc)).dia $\leq$ f-2 } \hfill //O(1)
\state res $\gets$ res + (siguiente(it)).cantA \hfill //O(1)
\endif
\state avanzar(itAcc) \hfill //O(1)
\endwhile
\EndFunction 
\end{algorithmic}
\hrule
\complejidad{1}
\end{algorithm}

\begin{algorithm}[H]
\caption{iEstaOrdenada?}
\begin{algorithmic}[1]
\Function{iestaOrdenada?}{\param{in}{it}{estrITPL}, \param{in}{f}{Fecha}}{$\ensuremath{\rightarrow}$ \param{}{res}{bool}}
\state res $\gets$ true \hfill //O(1)
\state nat aux $\gets$ cantAccesosDesde(it,f) \hfill //O(1)
\state avanzar(it) \hfill //O(1)
\while{haySiguiente(it)} \hfill //O(longitud(siguientes(it)))
\If{cantAccesosDesde(it,f) > aux} \hfill //O(1)
\state res $\gets$ false \hfill //O(1)
\endif
\state aux $\gets$ cantAccesosDesde(it,f) \hfill //O(1)
\state avanzar(it) \hfill //O(1)
\endwhile
\EndFunction 
\end{algorithmic}
\hrule
\complejidad{longitud(siguientes(it))}
\end{algorithm}
