{\param{in}{gr}{puntero(gremio)}, \param{in}{ps}{nat}, \param{in}{tp}{nat}}{estrP}
{
	\state res.gremio \asig gr										\inlineC{1}
	\state res.piso \asig ps										\inlineC{1}
	\state res.tope \asig tp										\inlineC{1}
}

{Crea una paritaria a partir de un puntero a un gremio, un valor piso y un valor tope.}
