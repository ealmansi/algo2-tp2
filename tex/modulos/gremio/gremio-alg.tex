\algoritmoI{NuevoGremio}
{\param{in}{es}{conj(empresa)}, \param{in}{\#as}{nat}}{estrG}
{
	\state res.empresas \asig es 							\inlineC{1}
	\state res.\#afiliados \asig \#as 						\inlineC{1}
	\state res.id \asig 0 									\inlineC{1}
}
{1}
{Crea un gremio a partir de un conjunto de empresas y un numero de afiliados.}

\algoritmoI{ObtenerEmpresas}
{\param{in}{gr}{estrG}}{conj(empresa)}
{
	\state res \asig gr.empresas 							\inlineC{1}
}
{1}
{Devuelve el conjunto de empresas del gremio}

\algoritmoI{Obtener\#Afiliados}
{\param{in}{gr}{estrG}}{nat}
{
	\state res \asig gr.\#afiliados 						\inlineC{1}
}
{1}
{Devuelve el numero de afiliados del gremio}

\algoritmoI{ObtenerId}
{\param{in}{gr}{estrG}}{nat}
{
	\state res \asig gr.id 									\inlineC{1}
}
{1}
{Devuelve el ID que identifica un\'ivocamente al gremio}

\algoritmoI{GuardarId}
{\param{in/out}{gr}{estrG}, \param{in}{id}{nat}}{}
{
	\state gr.id \asig id 									\inlineC{1}
}
{1}
{Asigna el ID \'unico al gremio}
